```python
import random
from collections import defaultdict

state = 1.0
goal = 80

symbols = list(range(1, 106))
memory = defaultdict(int)

# 🔁 параметри О (вони будуть вчитись)
chaos_weight = 1.0
balance_weight = 1.0

    global chaos_weight, balance_weight
    
    balance = ((x + y) / 2) * balance_weight
    chaos = random.uniform(-1, 1) * chaos_weight
    
    return balance + chaos

    key = round(value)
    memory[key] += 1
    return memory[key]

def goal_pull(x):
    return 1 / (1 + abs(goal - x))

def update_O(performance):
    global chaos_weight, balance_weight
    
    # якщо добре → зменшуємо хаос
    if performance > 0.5:
        chaos_weight *= 0.99
        balance_weight *= 1.01
    else:
        chaos_weight *= 1.01
        balance_weight *= 0.99

history = []

for step in range(10000):
    s = random.choice(symbols)
    
    new_state = O(state, s)
    
    strength = reinforce(new_state)
    attraction = goal_pull(new_state)
    
    # головна формула
    state = new_state * (1 + strength * 0.0003 + attraction * 0.05)
    
    performance = attraction  # наскільки близько до цілі
    
    update_O(performance)
    
    history.append(state)

print("final state:", state)
print("chaos_weight:", chaos_weight)
print("balance_weight:", balance_weight)

print("\nTop states:")
top = sorted(memory.items(), key=lambda x: -x[1])[:10]
for k,v in top:
    print(k, v)
```

```python
import random
from collections import defaultdict

# 🔢 параметри
NUM_AGENTS = 5
ENV_SIZE = 105

# 🎯 глобальна ціль (можна змінювати)
GOAL = 80

# 🌍 середовище
environment = [random.uniform(1, 105) for _ in range(ENV_SIZE)]

# 🤖 агенти (Я)
agents = [
    {
        "state": random.uniform(1, 105),
        "memory": defaultdict(int),
        "chaos": 1.0,
        "balance": 1.0
    }
    for _ in range(NUM_AGENTS)
]

def O(agent, x, y):
    balance = ((x + y) / 2) * agent["balance"]
    chaos = random.uniform(-1, 1) * agent["chaos"]
    return balance + chaos

def reinforce(agent, value):
    key = round(value)
    agent["memory"][key] += 1
    return agent["memory"][key]

def goal_pull(x):
    return 1 / (1 + abs(GOAL - x))

def update_O(agent, performance):
    if performance > 0.5:
        agent["chaos"] *= 0.99
        agent["balance"] *= 1.01
    else:
        agent["chaos"] *= 1.01
        agent["balance"] *= 0.99

def update_environment(env, agents):
    avg_state = sum(a["state"] for a in agents) / len(agents)
    for i in range(len(env)):
        influence = (avg_state - env[i]) * 0.01
        noise = random.uniform(-0.5, 0.5)
        env[i] += influence + noise

def create_new_symbols(env, agents):
    # 🔥 створення нових "символів"
    if random.random() < 0.05:
        new_symbol = sum(a["state"] for a in agents) / len(agents)
        env.append(new_symbol)

def interaction(a1, a2):
    # 🤝 або ⚔️ взаємодія
    diff = abs(a1["state"] - a2["state"])
    
    if diff < 5:
        # кооперація
        avg = (a1["state"] + a2["state"]) / 2
        a1["state"] = avg
        a2["state"] = avg
    else:
        # конкуренція
        if a1["state"] > a2["state"]:
            a2["state"] *= 0.99
        else:
            a1["state"] *= 0.99

# 🚀 симуляція
for step in range(10000):
    
    # кожен агент діє
    for agent in agents:
        symbol = random.choice(environment)
        
        new_state = O(agent, agent["state"], symbol)
        
        strength = reinforce(agent, new_state)
        attraction = goal_pull(new_state)
        
        agent["state"] = new_state * (1 + strength * 0.0003 + attraction * 0.05)
        
        update_O(agent, attraction)
    
    # 🤝 взаємодія агентів
    for i in range(len(agents)):
        for j in range(i+1, len(agents)):
            interaction(agents[i], agents[j])
    
    # 🌍 середовище змінюється
    update_environment(environment, agents)
    
    # 🌱 нові символи
    create_new_symbols(environment, agents)

# 📊 результат
print("=== AGENTS ===")
for i, a in enumerate(agents):
    print(f"Agent {i}: state={a['state']:.2f}, chaos={a['chaos']:.3f}, balance={a['balance']:.3f}")

print("\nEnvironment size:", len(environment))

print("\nTop memory patterns (Agent 0):")
top = sorted(agents[0]["memory"].items(), key=lambda x: -x[1])[:10]
for k,v in top:
    print(k, v)
```

# 🔍 Що ти зараз реально тестиш

# 🔥 Код: О еволюціонує

```python id="2qzv1p"
import random
from collections import defaultdict
import copy

NUM_AGENTS = 6
ENV_SIZE = 105
GOAL = 80

environment = [random.uniform(1, 105) for _ in range(ENV_SIZE)]

def create_agent():
    return {
        "state": random.uniform(1, 105),
        "memory": defaultdict(int),
        "dna": {
            "balance": random.uniform(0.5, 1.5),
            "chaos": random.uniform(0.5, 1.5),
            "memory": random.uniform(0.0001, 0.001),
            "goal": random.uniform(0.01, 0.1),
        },
        "score": 0
    }

agents = [create_agent() for _ in range(NUM_AGENTS)]

def O(agent, x, y):
    dna = agent["dna"]
    
    balance = ((x + y) / 2) * dna["balance"]
    chaos = random.uniform(-1, 1) * dna["chaos"]
    
    return balance + chaos

def reinforce(agent, value):
    key = round(value)
    agent["memory"][key] += 1
    return agent["memory"][key]

def goal_pull(x):
    return 1 / (1 + abs(GOAL - x))

def mutate(dna):
    new_dna = dna.copy()
    for k in new_dna:
        if random.random() < 0.3:
            new_dna[k] *= random.uniform(0.9, 1.1)
    return new_dna

def reproduce(agents):
    # 🔥 відбір + мутація
    agents.sort(key=lambda a: -a["score"])
    
    new_agents = agents[:3]  # виживають топ
    
    while len(new_agents) < NUM_AGENTS:
        parent = random.choice(new_agents)
        child = copy.deepcopy(parent)
        child["dna"] = mutate(parent["dna"])
        child["state"] = random.uniform(1, 105)
        child["memory"] = defaultdict(int)
        child["score"] = 0
        new_agents.append(child)
    
    return new_agents

# 🚀 симуляція
for epoch in range(50):
    
    for agent in agents:
        agent["score"] = 0
    
    for step in range(2000):
        for agent in agents:
            symbol = random.choice(environment)
            
            new_state = O(agent, agent["state"], symbol)
            
            strength = reinforce(agent, new_state)
            attraction = goal_pull(new_state)
            
            agent["state"] = new_state * (
                1 +
                strength * agent["dna"]["memory"] +
                attraction * agent["dna"]["goal"]
            )
            
            agent["score"] += attraction
    
    # 🧬 еволюція
    agents = reproduce(agents)

# 📊 результат
print("=== FINAL AGENTS ===")
for i, a in enumerate(agents):
    print(f"\nAgent {i}")
    print("state:", round(a["state"], 2))
    print("score:", round(a["score"], 2))
    print("dna:", {k: round(v,4) for k,v in a["dna"].items()})
```

# 🧬 Код: О створює свої цілі

```python
import random
from collections import defaultdict
import copy

NUM_AGENTS = 6
ENV_SIZE = 105

environment = [random.uniform(1, 105) for _ in range(ENV_SIZE)]

def create_agent():
    return {
        "state": random.uniform(1, 105),
        "memory": defaultdict(int),
        "dna": {
            "balance": random.uniform(0.5, 1.5),
            "chaos": random.uniform(0.5, 1.5),
            "memory": random.uniform(0.0001, 0.001),
            "goal_weight": random.uniform(0.01, 0.1),
        },
        "goal": random.uniform(1, 105),
        "score": 0
    }

agents = [create_agent() for _ in range(NUM_AGENTS)]

def O(agent, x, y):
    dna = agent["dna"]
    balance = ((x + y) / 2) * dna["balance"]
    chaos = random.uniform(-1, 1) * dna["chaos"]
    return balance + chaos

def reinforce(agent, value):
    key = round(value)
    agent["memory"][key] += 1
    return agent["memory"][key]

def goal_pull(agent, x):
    return 1 / (1 + abs(agent["goal"] - x))

def mutate(dna):
    new_dna = dna.copy()
    for k in new_dna:
        if random.random() < 0.3:
            new_dna[k] *= random.uniform(0.9, 1.1)
    return new_dna

def update_goal(agent):
    # 🔥 агент змінює своє "питання"
    
    # 1. бере популярні стани (памʼять)
    if agent["memory"]:
        best_states = sorted(agent["memory"].items(), key=lambda x: -x[1])[:5]
        candidate = random.choice(best_states)[0]
    else:
        candidate = random.uniform(1, 105)
    
    # 2. додає трохи хаосу
    candidate += random.uniform(-5, 5)
    
    # 3. зміщення до середовища
    env_influence = random.choice(environment)
    
    # новий goal = баланс
    agent["goal"] = (candidate + env_influence) / 2

def reproduce(agents):
    agents.sort(key=lambda a: -a["score"])
    new_agents = agents[:3]
    
    while len(new_agents) < NUM_AGENTS:
        parent = random.choice(new_agents)
        child = copy.deepcopy(parent)
        child["dna"] = mutate(parent["dna"])
        child["state"] = random.uniform(1, 105)
        child["memory"] = defaultdict(int)
        child["score"] = 0
        child["goal"] = random.uniform(1, 105)
        new_agents.append(child)
    
    return new_agents

# 🚀 симуляція
for epoch in range(50):
    
    for agent in agents:
        agent["score"] = 0
    
    for step in range(2000):
        for agent in agents:
            symbol = random.choice(environment)
            
            new_state = O(agent, agent["state"], symbol)
            
            strength = reinforce(agent, new_state)
            attraction = goal_pull(agent, new_state)
            
            agent["state"] = new_state * (
                1 +
                strength * agent["dna"]["memory"] +
                attraction * agent["dna"]["goal_weight"]
            )
            
            agent["score"] += attraction
            
            # 🔥 головне: іноді міняє goal
            if random.random() < 0.01:
                update_goal(agent)
    
    agents = reproduce(agents)

# 📊 результат
print("=== FINAL AGENTS ===")
for i, a in enumerate(agents):
    print(f"\nAgent {i}")
    print("state:", round(a["state"], 2))
    print("goal:", round(a["goal"], 2))
    print("score:", round(a["score"], 2))
    print("dna:", {k: round(v,4) for k,v in a["dna"].items()})
```
